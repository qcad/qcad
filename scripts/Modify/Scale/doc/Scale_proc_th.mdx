import KeepProportionsOn from './KeepProportionsOn.png';
import ByMouse from './ByMouse.png';

1. เลือกเอนทิตีที่คุณต้องการปรับสเกล
2. เริ่มเครื่องมือนี้
3. กำหนดจุดศูนย์กลางของการปรับสเกลด้วยเมาส์หรือป้อนพิกัดในบรรทัดคำสั่ง
4. กล่องโต้ตอบสเกลจะแสดงขึ้นเพื่อให้คุณป้อนตัวคูณสเกล  
   หากคุณต้องการปรับสเกลด้วยตัวคูณสองค่าที่แตกต่างกันในทิศทาง X และ Y ให้ยกเลิกการทำเครื่องหมายที่ปุ่มสำหรับการปรับสเกลตามสัดส่วน:  
   <Image src={KeepProportionsOn} alt="KeepProportionsOn" width={21} height={40} />  
   จากนั้นคุณสามารถป้อนตัวคูณสเกลสองค่าที่แตกต่างกันสำหรับ X และ Y  
   หากคุณต้องการปรับสเกลการเลือกด้วยเมาส์ ให้ทำเครื่องหมายที่ปุ่มเคอร์เซอร์เมาส์:  
   <Image src={ByMouse} alt="ByMouse" width={21} height={40} />
5. คลิก „OK"
6. หากคุณเลือกที่จะปรับสเกลการเลือกด้วยเมาส์ก่อนหน้านี้ ตอนนี้คุณต้องระบุจุดอ้างอิงและจุดเป้าหมายสำหรับการดำเนินการปรับสเกล
