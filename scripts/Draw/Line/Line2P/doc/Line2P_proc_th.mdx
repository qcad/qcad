import Close from './Close.png';
import Undo from './Undo.png';

1. เลือก [ประเภทเส้น](/th/manual/scripts/Draw/Line/doc/Line#type) ที่ต้องการในแถบเครื่องมือตัวเลือก
2. ระบุจุดเริ่มต้นของเซกเมนต์เส้นแรก คุณสามารถใช้เมาส์หรือป้อนพิกัดในคอนโซล
3. ระบุจุดสิ้นสุดของเซกเมนต์เส้นแรก
4. ระบุจุดสิ้นสุดของเซกเมนต์เส้นเพิ่มเติม คลิกปุ่ม „Close" ในแถบเครื่องมือตัวเลือกเพื่อปิดลำดับ:
   <Image src={Close} alt="Close" width={40} height={40} />  
   หากคุณต้องการเลิกทำเซกเมนต์เส้นเดียว คุณสามารถทำได้โดยการคลิกปุ่ม „Undo":
   <Image src={Undo} alt="Undo" width={40} height={40} />
5. คุณสามารถกำหนดความยาวหรือมุมของเส้นเป็นค่าคงที่โดยใช้ช่องป้อนที่เหมาะสมในแถบเครื่องมือตัวเลือก
